generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider  = "postgresql"
  url       = env("DATABASE_URL")
  directUrl = env("DIRECT_URL")
}

model User {
  id             String          @id(map: "user_profiles_pkey") @default(uuid()) @db.Uuid
  username       String          @unique(map: "user_profiles_username_key")
  passwordHash   String          @map("password_hash")
  role           UserRole        @default(SPPG)
  schoolId       String?         @unique @map("school_id") @db.Uuid
  createdAt      DateTime        @default(now()) @map("created_at")
  updatedAt      DateTime        @updatedAt @map("updated_at")
  petugasBatches BatchProduksi[] @relation("PetugasBatch")
  sekolahBatches BatchSekolah[]  @relation("SekolahBatch")
  managedSchools School[]        @relation("SppgSchools")
  school         School?         @relation("SchoolAccount", fields: [schoolId], references: [id])

  @@map("users")
}

model School {
  id                   String                    @id @default(uuid()) @db.Uuid
  name                 String
  npsn                 String?
  address              String?
  sppgId               String                    @map("sppg_id") @db.Uuid
  createdAt            DateTime                  @default(now()) @map("created_at")
  updatedAt            DateTime                  @updatedAt @map("updated_at")
  distributionReceipts BatchDistributionSchool[]
  progress             SchoolProgress?
  sppg                 User                      @relation("SppgSchools", fields: [sppgId], references: [id])
  account              User?                     @relation("SchoolAccount")

  @@index([sppgId])
  @@map("schools")
}

model SchoolProgress {
  id        String               @id @default(uuid()) @db.Uuid
  schoolId  String               @unique @map("school_id") @db.Uuid
  status    SchoolProgressStatus @default(BELUM_ADA)
  notes     String?
  updatedAt DateTime             @updatedAt @map("updated_at")
  createdAt DateTime             @default(now()) @map("created_at")
  school    School               @relation(fields: [schoolId], references: [id], onDelete: Cascade)

  @@map("school_progress")
}

model BatchProduksi {
  id               String              @id
  menuId           String              @db.Uuid
  totalPorsi       Int
  waktuMulai       DateTime?
  waktuSelesai     DateTime?
  petugasId        String?             @db.Uuid
  driverId         String?             @db.Uuid
  noKendaraan      String?
  ruteDistribusi   String?
  jamKeberangkatan DateTime?
  status           BatchStatus         @default(DRAFT)
  catatanKualitas  String?
  isApproved       Boolean             @default(false)
  createdAt        DateTime            @default(now()) @map("created_at")
  updatedAt        DateTime            @updatedAt @map("updated_at")
  distributions    BatchDistribution[]
  foto             BatchFoto[]
  driver           Driver?             @relation("DriverBatch", fields: [driverId], references: [id])
  menu             MenuMaster          @relation(fields: [menuId], references: [id])
  petugas          User?               @relation("PetugasBatch", fields: [petugasId], references: [id])
  sekolah          BatchSekolah[]
  varian           BatchVarian[]

  @@map("batch_produksi")
}

model BatchDistribution {
  id         String                    @id @default(uuid()) @db.Uuid
  batchId    String                    @map("batch_id")
  waktuKirim DateTime?                 @map("waktu_kirim")
  status     BatchDistributionStatus   @default(DIKIRIM)
  createdAt  DateTime                  @default(now()) @map("created_at")
  updatedAt  DateTime                  @updatedAt @map("updated_at")
  schools    BatchDistributionSchool[]
  batch      BatchProduksi             @relation(fields: [batchId], references: [id], onDelete: Cascade)

  @@index([batchId])
  @@map("batch_distributions")
}

model BatchDistributionSchool {
  id             String                        @id @default(uuid()) @db.Uuid
  distributionId String                        @map("distribution_id") @db.Uuid
  schoolId       String                        @map("school_id") @db.Uuid
  jumlahPorsi    Int                           @map("jumlah_porsi")
  status         BatchDistributionSchoolStatus @default(MENUNGGU)
  receivedAt     DateTime?                     @map("received_at")
  rejectedReason String?                       @map("rejected_reason")
  createdAt      DateTime                      @default(now()) @map("created_at")
  updatedAt      DateTime                      @updatedAt @map("updated_at")
  distribution   BatchDistribution             @relation(fields: [distributionId], references: [id], onDelete: Cascade)
  school         School                        @relation(fields: [schoolId], references: [id], onDelete: Cascade)

  @@unique([distributionId, schoolId])
  @@index([schoolId])
  @@map("batch_distribution_schools")
}

model Batch {

model Distribution {
  id               String             @id @default(uuid())
  batchId          String
  sekolahId        String
  waktuKirim       DateTime?          @map("waktu_kirim")
  jumlahPorsi      Int

  createdAt        DateTime           @default(now()) @map("created_at")
  updatedAt        DateTime           @updatedAt @map("updated_at")

  receivingReceipt ReceivingReceipt?

  @@map("distributions")
}
